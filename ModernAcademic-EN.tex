\section{Quick Start}
% ============================================================================

To use this template, create a new \texttt{.tex} file with the following minimal structure:

\begin{lstlisting}[language=TeX,caption={Minimal document structure}]
\documentclass[singlecolumn]{ModernAcademic-EN}

\title{Your Paper Title}
\author{Author Name}
\abstract{Your abstract text...}

\begin{document}
\maketitle
\section{Introduction}
Your content here...
\bibliography{references}
\end{document}
\end{lstlisting}

\subsection{Class Options}
The class supports two layout options:
\begin{itemize}
    \item \texttt{singlecolumn} (default) --- Standard single-column layout with 11pt font
    \item \texttt{twocolumn} --- Two-column layout with 10pt font for conference papers
\end{itemize}

\section{Theme Customization}
\label{sec:theme}
% ============================================================================

\subsection{Primary Theme Color}
Use \verb|\setthemecolor{color}| to change the primary accent color used for titles, section headings, and links. The color can be specified in hexadecimal or RGB format:

\begin{lstlisting}[language=TeX]
\setthemecolor{1a2a6c}      % Hex format (default blue)
\setthemecolor{185,28,28}   % RGB format (red)
\end{lstlisting}

\subsection{Additional Color Settings}
Fine-tune individual element colors:
\begin{lstlisting}[language=TeX]
\setlinkcolor{0000FF}   % Internal links
\setcitecolor{2d5a27}   % Citation links
\seturlcolor{5c3566}    % External URL links
\end{lstlisting}

\label{sec:math}

The template loads \texttt{amsmath}, \texttt{amssymb}, and \texttt{amsthm} for comprehensive math support.

\subsection{Equations}
Inline math: $E = mc^2$, $\sum_{i=1}^{n} x_i$. Display equations with automatic numbering:

\begin{equation}
    \mathcal{L}(\theta) = \mathbb{E}_{x \sim p_{\text{data}}} \left[ \log p_\theta(x) \right]
    \label{eq:likelihood}
\end{equation}

Aligned equations:
\begin{align}
    H(X)   & = -\sum_{x} p(x) \log p(x) \label{eq:entropy} \\
    I(X;Y) & = H(X) - H(X|Y) \label{eq:mi}
\end{align}

Cross-reference: \cref{eq:entropy} shows the entropy definition.

\subsection{Pseudocode}
\begin{algorithm}[htbp]
    \caption{Example algorithm using \texttt{algpseudocode}}
    \label{alg:example}
    \begin{algorithmic}[1]
        \Procedure{TrainModel}{$\mathcal{D}, \theta_0, \eta$}
        \State $\theta \gets \theta_0$
        \While{not converged}
        \State $\mathcal{B} \gets$ \Call{SampleBatch}{$\mathcal{D}$}
        \State $g \gets \nabla_\theta \mathcal{L}(\theta; \mathcal{B})$
        \State $\theta \gets \theta - \eta \cdot g$
        \EndWhile
        \State \Return $\theta$
        \EndProcedure
    \end{algorithmic}
\end{algorithm}

\subsection{Code Listings}
\begin{lstlisting}[language=Python,caption={Python code example}]
def train_step(model, batch, optimizer):
    loss = model.compute_loss(batch)
    loss.backward()
    optimizer.step()
    return loss.item()
\end{lstlisting}
