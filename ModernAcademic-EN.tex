\section{Quick Start}
\label{sec:quick-start}
% ============================================================================

The class is designed as a compact academic-paper template for pdf\LaTeX. A minimal document needs a class declaration, document metadata, \verb|\maketitle|, and normal article content.

\begin{lstlisting}[language=TeX,caption={Minimal single-column document},label={lst:minimal}]
\documentclass[singlecolumn]{ModernAcademic-EN}

\title{Your Paper Title}
\author{Author Name}
\shorttitle{Short Title}
\shortauthor{Author}
\affiliation{Institution Name}
\email{author@example.org}
\abstract{A concise abstract.}

\begin{document}
\maketitle
\section{Introduction}
\lipsum[1]
\bibliography{references}
\end{document}
\end{lstlisting}

\subsection{Class Options}
\label{subsec:class-options}

\begin{itemize}
  \item \texttt{singlecolumn}: the default article layout, using an 11 pt base font.
  \item \texttt{twocolumn}: a compact two-column layout, using a 10 pt base font and the class-defined column separation.
  \item Standard \texttt{article} options not handled by the class are passed through to the base class.
\end{itemize}

\subsection{Compilation}

The English template is configured for pdf\LaTeX. A bibliography run is needed when references are used.

\begin{lstlisting}[language=bash,caption={Typical compilation sequence}]
pdflatex ModernAcademic-EN.tex
bibtex ModernAcademic-EN
pdflatex ModernAcademic-EN.tex
pdflatex ModernAcademic-EN.tex
\end{lstlisting}

\subsection{Abstract Footnotes}

Abstract-box footnotes are promoted to normal page footnotes by default. In that mode, a \verb|\footnote| written inside \verb|\abstract| is collected from the \texttt{tcolorbox} title block and typeset in the regular page footnote area; the note in this sample abstract demonstrates that behavior.

If the document should keep abstract notes visually inside the abstract box, use the following command in the preamble before \verb|\begin{document}|:

\begin{lstlisting}[language=TeX]
\disableabstractfootnotes
\end{lstlisting}

After disabling promotion, footnotes written in the abstract remain at the bottom of the abstract box. This changes only the abstract box behavior; ordinary body footnotes still use the normal page footnote area.

\subsection{Theorem Environments}

\begin{definition}[Convex function]
  A function $f:\mathbb{R}^n\to\mathbb{R}$ is convex if, for all $x,y\in\mathbb{R}^n$ and $\lambda\in[0,1]$,
  \[
    f(\lambda x + (1-\lambda)y)
    \leq \lambda f(x) + (1-\lambda)f(y).
  \]
  \label{def:convex}
\end{definition}

\begin{theorem}[Jensen's inequality]
  Let $f$ be convex and let $X$ be an integrable random variable. Then
  \[
    f(\mathbb{E}X) \leq \mathbb{E}f(X).
  \]
  \label{thm:jensen}
\end{theorem}

\begin{proof}
  The statement follows by applying the supporting-hyperplane characterization of convex functions to \cref{def:convex} and then taking expectations. \lipsum[9][1-2]
\end{proof}

\begin{lemma}
  If a sequence $(a_n)$ is nonnegative and summable, then $a_n\to0$.
  \label{lem:summable}
\end{lemma}

\begin{proposition}
  The cross-reference names for \cref{thm:jensen,lem:summable} are produced by \textnormal{\texttt{cleveref}}.
\end{proposition}

\begin{corollary}
  Any finite nonnegative sequence satisfies the conclusion of \cref{lem:summable} after extension by zeros.
\end{corollary}

\begin{example}
  The quadratic function $f(x)=x^\top x$ is a standard convex function.
\end{example}

\begin{remark}
  The class provides numbered theorem, lemma, corollary, proposition, definition, and example environments, plus an unnumbered remark environment.
\end{remark}

\section{Algorithms and Code Listings}
\label{sec:algorithms}
% ============================================================================

The class does not force an algorithm or code package. This example loads \texttt{algorithm}, \texttt{algpseudocode}, and \texttt{listings} in the document preamble so authors can choose their preferred tools.

\begin{algorithm}[htbp]
  \caption{Template-aware experiment loop}
  \label{alg:example}
  \begin{algorithmic}[1]
    \Procedure{RunStudy}{$\mathcal{D}, \Theta, K$}
      \For{$k \gets 1$ to $K$}
        \State $\theta_k \gets$ \Call{Initialize}{$\Theta$}
        \State $\ell_k \gets$ \Call{TrainAndEvaluate}{$\mathcal{D}, \theta_k$}
      \EndFor
      \State \Return $\arg\min_k \ell_k$
    \EndProcedure
  \end{algorithmic}
\end{algorithm}

\begin{lstlisting}[language=Python,caption={Small Python listing with the class color theme},label={lst:python}]
def summarize(scores):
    total = sum(scores)
    mean = total / len(scores)
    return {"count": len(scores), "mean": mean}
\end{lstlisting}
