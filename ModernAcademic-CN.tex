\section{快速开始}
% ============================================================================

使用此模板，创建一个新的 \texttt{.tex} 文件，包含以下最小结构：

\begin{lstlisting}[language=TeX,caption={最小文档结构}]
\documentclass[singlecolumn]{ModernAcademic-CN}

\title{您的论文标题}
\author{作者姓名}
\abstract{摘要内容...}

\begin{document}
\maketitle
\section{引言}
正文内容...
\zhlipsum[1]
\bibliography{references}
\end{document}
\end{lstlisting}

\subsection{类选项}
类文件支持两种布局选项：
\begin{itemize}
    \item \texttt{singlecolumn}（默认）——标准单栏布局，11pt 字体
    \item \texttt{twocolumn}——双栏布局，10pt 字体
\end{itemize}

\subsection{主题色}
使用 \verb|\setthemecolor{颜色}| 更改标题、章节标题和链接使用的主色调。
颜色可以用十六进制或RGB格式指定：

\begin{lstlisting}[language=TeX]
\setthemecolor{1a2a6c}      % 十六进制格式（默认蓝色）
\setthemecolor{185,28,28}   % RGB格式（红色）
\end{lstlisting}

\subsection{其他颜色设置}
可以单独调整各元素的颜色：
\begin{lstlisting}[language=TeX]
\setlinkcolor{0000FF}   % 内部链接
\setcitecolor{2d5a27}   % 引用链接
\seturlcolor{5c3566}    % 外部URL链接
\end{lstlisting}

\subsection{公式}
行内公式：$E = mc^2$，$\sum_{i=1}^{n} x_i$。
行间公式自动编号：

\begin{equation}
    \mathcal{L}(\theta) = \mathbb{E}_{x \sim p_{\text{data}}} \left[ \log p_\theta(x) \right]
    \label{eq:likelihood}
\end{equation}

对齐公式：
\begin{align}
    H(X)   & = -\sum_{x} p(x) \log p(x) \label{eq:entropy} \\
    I(X;Y) & = H(X) - H(X|Y) \label{eq:mi}
\end{align}

交叉引用：\cref{eq:entropy}展示了熵的定义。

\subsection{伪代码}
\begin{algorithm}[htbp]
    \caption{使用 \texttt{algpseudocode} 的示例算法}
    \label{alg:example}
    \begin{algorithmic}[1]
        \Procedure{训练模型}{$\mathcal{D}, \theta_0, \eta$}
        \State $\theta \gets \theta_0$
        \While{未收敛}
        \State $\mathcal{B} \gets$ \Call{采样批次}{$\mathcal{D}$}
        \State $g \gets \nabla_\theta \mathcal{L}(\theta; \mathcal{B})$
        \State $\theta \gets \theta - \eta \cdot g$
        \EndWhile
        \State \Return $\theta$
        \EndProcedure
    \end{algorithmic}
\end{algorithm}

def train_step(model, batch, optimizer):
    loss = model.compute_loss(batch)
    loss.backward()
    optimizer.step()
    return loss.item()
