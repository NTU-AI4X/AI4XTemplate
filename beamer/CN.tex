\begin{academicframe}[当前主题中的 Beamer 核心排版单元]{Columns 与宽度}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.47\textwidth}
      \begin{block}{双栏模式}
        \scriptsize
        \begin{itemize}
          \item 用 \texttt{\textbackslash begin\{columns\}[T,onlytextwidth]} 开始。
          \item 用 \texttt{\textbackslash begin\{column\}\{0.48\textbackslash textwidth\}} 添加栏。
          \item 预留栏间距后，总宽度不要超过 \texttt{1.0\textbackslash textwidth}。
          \item 选项 \texttt{T} 用于顶部对齐。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.47\textwidth}
      \begin{exampleblock}{何时使用 columns}
        \scriptsize
        方法与结果对比、图文并排说明、左右两组命令示例，都适合使用
        \texttt{columns}。如果只是在单栏内部做局部对齐，用
        \texttt{minipage}；如果是整页结构，用 \texttt{columns}。
      \end{exampleblock}
    \end{column}
  \end{columns}

  \vspace{0.5em}
  \begin{alertblock}{实用规则}
    \scriptsize
    优先使用 \texttt{onlytextwidth}，这样内容会与主题 body 区域对齐，
    而不是按整张纸面宽度排版。
  \end{alertblock}
\end{academicframe}

\begin{academicframe}[定理与 proof 环境会编号]{数学}
  \begin{theorem}[率失真目标]
    \normalfont
    设 $x$ 为源样本，$\hat{x}$ 为其重建结果。学习式压缩中的典型
    优化目标可写为
    \[
      \mathcal{L}(\theta)=
      \mathbb{E}_{x\sim p_{\mathrm{data}}}
      \left[R_{\theta}(x)+\lambda d(x,\hat{x})\right].
    \]
  \end{theorem}

  \begin{proof}
    \normalfont
    该表达式结合了期望码率项与失真惩罚项；
    $\lambda$ 用于选择率失真曲线上的工作点。
  \end{proof}
\end{academicframe}
