\begin{academicframe}[Core Beamer layout primitives inside this theme]{Columns and Widths}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.47\textwidth}
      \begin{block}{Two-column pattern}
        \scriptsize
        \begin{itemize}
          \item Start with \texttt{\textbackslash begin\{columns\}[T,onlytextwidth]}.
          \item Add panels with \texttt{\textbackslash begin\{column\}\{0.48\textbackslash textwidth\}}.
          \item Keep column widths below \texttt{1.0\textbackslash textwidth} after gaps.
          \item Use \texttt{T} to align column tops.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.47\textwidth}
      \begin{exampleblock}{When to use columns}
        \scriptsize
        Use columns for method/result comparisons, figure-plus-explanation
        slides, or side-by-side command examples. For small local alignment
        inside one column, use \texttt{minipage}; for full slide structure, use
        \texttt{columns}.
      \end{exampleblock}
    \end{column}
  \end{columns}

  \vspace{0.5em}
  \begin{alertblock}{Practical rule}
    \scriptsize
    Prefer \texttt{onlytextwidth}; it keeps content aligned with the theme body
    region instead of the full paper width.
  \end{alertblock}
\end{academicframe}

\begin{academicframe}[Theorem and proof environments are numbered]{Mathematics}
  \begin{theorem}[Rate-distortion objective]
    Let $x$ be a source sample and $\hat{x}$ its reconstruction. A typical
    learned compression objective can be written as
    \[
      \mathcal{L}(\theta)=
      \mathbb{E}_{x\sim p_{\mathrm{data}}}
      \left[R_{\theta}(x)+\lambda d(x,\hat{x})\right].
    \]
  \end{theorem}

  \begin{proof}
    The expression combines an expected bitrate term with a distortion penalty;
    $\lambda$ selects the operating point on the rate-distortion curve.
  \end{proof}
\end{academicframe}
